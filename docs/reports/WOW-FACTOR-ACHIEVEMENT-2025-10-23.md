```javascript

  DataVisualizationEngine,
} = require("./node0/data-visualization-engine");
const { StorytellingDashboard } = require("./node0/storytelling-dashboard");

// Initialize systems
const viz = new DataVisualizationEngine({ language: "en" });
const story = new StorytellingDashboard({ language: "en" });

// Update with real metrics
const metrics = {
  tps: 847.23,
  finality: 247,
  uptime: 99.87,
  blockHeight: 150342,
  peers: 34,
  mempool: 1847,
  validators: [
    { name: "Genesis-Validator", health: 97 },
    { name: "Node-Alpha", health: 94 },
  ],
};

// Update both systems
viz.updateMetrics(metrics);
const narrative = story.updateMetrics(metrics);

// Render visualization
viz.renderDashboard();

// Render narrative
story.renderNarrative(narrative);
```
